// Quick Sort Example
void quick(int a[], int lb, int ub) {
    int i, j, t, key;
    if(lb > ub) return;
    
    i = lb;
    j = ub;
    key = lb;
    
    while(i < j) {
        while(a[key] > a[i]) i++;
        while(a[key] < a[j]) j--;
        
        if(i < j) {
            t = a[i];
            a[i] = a[j];
            a[j] = t;
        }
    }
    
    // Recursively sort partitions
    quick(a, 0, j-1);
    quick(a, j+1, ub);
}
